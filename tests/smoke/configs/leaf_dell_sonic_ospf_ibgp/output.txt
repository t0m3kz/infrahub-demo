! Hostname configuration
hostname dc1-leaf-01

! Enable required features
nv overlay evpn
feature ospf
feature bgp
feature interface-vlan
feature vn-segment-vlan-based
feature nv overlay
vlan 100
  name Customer-A
  vn-segment 100

vlan 200
  name Customer-A
  vn-segment 200

interface Ethernet1
  description to spine-01
  ip address 10.1.0.1/31
  ip router ospf 1 area 0.0.0.0
  no shutdown
!

interface Ethernet2
  description to spine-02
  ip address 10.1.0.3/31
  ip router ospf 1 area 0.0.0.0
  no shutdown
!

interface Ethernet10
  description to server-01
  no shutdown
!

interface Loopback0
  description Router ID
  ip address 10.0.0.1/32
  no shutdown
!

router ospf 1
  router-id 10.0.0.1
  auto-cost reference-bandwidth 100000
  passive-interface default
  no passive-interface Ethernet1
  no passive-interface Ethernet2
!


router bgp 65000
  router-id 10.0.0.1
  no bgp default ipv4-unicast
  distance bgp 20 200 200
  bgp graceful-restart
  maximum-paths 4 ecmp 64
  !
  neighbor EVPN-PEERS peer-group
  neighbor EVPN-PEERS remote-as 65000
  neighbor EVPN-PEERS update-source Loopback0
  neighbor EVPN-PEERS send-community extended
  neighbor EVPN-PEERS maximum-routes 0
  neighbor 10.0.0.100 peer-group EVPN-PEERS
  neighbor 10.0.0.100 description spine-01
  neighbor 10.0.0.101 peer-group EVPN-PEERS
  neighbor 10.0.0.101 description spine-02
  !
  address-family ipv4
    redistribute connected route-map RM-CONN-2-BGP
  !
  address-family evpn
    neighbor EVPN-PEERS activate
!


!
! VXLAN Configuration with Microsegmentation (VRF-aware VNIs)
!
! Microsegmentation enabled: 1 VRF(s) for tenant isolation
!
interface vtep
  source Loopback0
  !
  ! L2 VNI Mappings (VLAN segments)
  map vni 10100 vlan 100
  !
  ! L3 VNI Mappings (VRF segments for microsegmentation)
  map vni 50001 vrf VRF_A
  no shutdown
!
!
! VRF Definitions (Microsegmentation)
!
vrf VRF_A
  description Tenant VRF for microsegmentation
!

! Configure route-maps
route-map RM-CONN-2-BGP permit 10
  match interface Loopback0
!
! VLAN SVI Configuration
!
interface Vlan100
  description Customer-A
  vrf forwarding VRF_A
  ip address 10.100.0.1/24
  no shutdown
!
interface Vlan200
  description Customer-A
  ip address 10.200.0.1/24
  no shutdown
!
!
! VRF Default Routes — firewall gateway per namespace (Option A: FW as inter-VRF router)
!
ip route vrf VRF_A 0.0.0.0/0 10.100.0.254
!

! End of configuration
