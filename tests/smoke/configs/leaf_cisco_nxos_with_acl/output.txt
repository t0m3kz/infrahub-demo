! dc1-leaf-01 Configuration
!
hostname dc1-leaf-01
!
feature ospf
feature bgp
feature interface-vlan
feature lacp
feature vpc
feature lldp
!

! VLANs
vlan 100
  name Customer-A
!
vlan 200
  name Customer-A
!
!
! Zero-Trust ACLs (one per segment VLAN — inbound on SVI)
!
ip access-list ACL-VLAN100-IN
  10 permit tcp any any eq 443 ! allow-https
  20 permit tcp any any eq 80 ! allow-http
  9990 deny ip any any log ! implicit-deny-all
!
ip access-list ACL-VLAN200-IN
  10 permit tcp any any eq 22 ! allow-ssh
  9990 deny ip any any log ! implicit-deny-all
!
!
! Interfaces
interface Ethernet1
  no shutdown
  ip address 10.1.0.1/31

!
interface Ethernet2
  no shutdown
  ip address 10.1.0.3/31

!
interface Ethernet10
  no shutdown

!
interface Loopback0
  no shutdown
  ip address 10.0.0.1/32

! OSPF Configuration

! BGP Configuration

router bgp 65000
  router-id 10.0.0.1
  graceful-restart
  !
  template peer-policy EVPN-PEERS-POLICY
    send-community
    send-community extended
  !
  template peer-session EVPN-PEERS-SESSION
    update-source loopback0
  !
  address-family ipv4 unicast
    redistribute direct route-map RM-CONN-2-BGP
    maximum-paths 8
  !
  address-family l2vpn evpn
  !
  neighbor 10.0.0.100
    remote-as 65000
    description spine-01
    inherit peer-session EVPN-PEERS-SESSION
    address-family l2vpn evpn
      inherit peer-policy EVPN-PEERS-POLICY
    !
  neighbor 10.0.0.101
    remote-as 65000
    description spine-02
    inherit peer-session EVPN-PEERS-SESSION
    address-family l2vpn evpn
      inherit peer-policy EVPN-PEERS-POLICY
    !
!
router bgp 65001
  router-id 10.0.0.1
  graceful-restart
  !
  template peer-policy UNDERLAY-PEERS-POLICY
    send-community
    send-community extended
  !
  template peer-session UNDERLAY-PEERS-SESSION
    bfd
  !
  address-family ipv4 unicast
    redistribute direct route-map RM-CONN-2-BGP
    maximum-paths 8
  !
  address-family l2vpn evpn
  !
  neighbor 10.1.0.0
    remote-as 65100
    description spine-01
    inherit peer-session UNDERLAY-PEERS-SESSION
    address-family ipv4 unicast
      inherit peer-policy UNDERLAY-PEERS-POLICY
    !
  neighbor 10.1.0.2
    remote-as 65101
    description spine-02
    inherit peer-session UNDERLAY-PEERS-SESSION
    address-family ipv4 unicast
      inherit peer-policy UNDERLAY-PEERS-POLICY
    !
