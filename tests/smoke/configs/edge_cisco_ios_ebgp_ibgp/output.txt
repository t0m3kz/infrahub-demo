! dc1-edge-01 Configuration
!
hostname dc1-edge-01
!
feature ospf
feature bgp
feature interface-vlan
feature lacp
feature vpc
feature lldp
!
!
! Interfaces
interface Ethernet1
  description to spine-01
  no shutdown
  ip address 10.1.0.1/31

!
interface Ethernet2
  description to spine-02
  no shutdown
  ip address 10.1.0.3/31

!
interface Loopback0
  description Router ID
  no shutdown
  ip address 10.0.0.1/32

!
!
!
! BGP Configuration

router bgp 65000
  bgp router-id 10.0.0.1
  bgp graceful-restart
  maximum-paths 4
  !
  neighbor EVPN-PEERS peer-group
  neighbor EVPN-PEERS remote-as 65000
  neighbor EVPN-PEERS update-source Loopback0
  neighbor EVPN-PEERS send-community both
  neighbor 10.0.0.100 peer-group EVPN-PEERS
  neighbor 10.0.0.100 description spine-01
  neighbor 10.0.0.101 peer-group EVPN-PEERS
  neighbor 10.0.0.101 description spine-02
  !
  address-family ipv4 unicast
    redistribute connected route-map RM-CONN-2-BGP
  exit-address-family
  !
  address-family l2vpn evpn
    neighbor EVPN-PEERS activate
  exit-address-family
!
router bgp 65001
  bgp router-id 10.0.0.1
  bgp graceful-restart
  maximum-paths 4
  !
  neighbor UNDERLAY-PEERS peer-group
  neighbor UNDERLAY-PEERS send-community both
  neighbor UNDERLAY-PEERS fall-over bfd
  neighbor 10.1.0.0 peer-group UNDERLAY-PEERS
  neighbor 10.1.0.0 remote-as 65100
  neighbor 10.1.0.0 description spine-01
  neighbor 10.1.0.2 peer-group UNDERLAY-PEERS
  neighbor 10.1.0.2 remote-as 65101
  neighbor 10.1.0.2 description spine-02
  !
  address-family ipv4 unicast
    redistribute connected route-map RM-CONN-2-BGP
    neighbor UNDERLAY-PEERS activate
  exit-address-family
!

!
! Route-maps
route-map RM-CONN-2-BGP permit 10
  match interface Loopback0
!
