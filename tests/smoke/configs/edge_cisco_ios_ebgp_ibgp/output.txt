!
! dc1-edge-01 Configuration
!
hostname dc1-edge-01
!
feature ospf
feature bgp
feature interface-vlan
feature lacp
feature vpc
feature lldp
!
!
! Interfaces
interface Ethernet1
  description to spine-01
  no shutdown
  ip address 10.1.0.1/31

!
interface Ethernet2
  description to spine-02
  no shutdown
  ip address 10.1.0.3/31

!
interface Loopback0
  description Router ID
  no shutdown
  ip address 10.0.0.1/32

!
!
!
! BGP Configuration

router bgp 65000
  bgp router-id 10.0.0.1
  bgp graceful-restart
  maximum-paths 4
  !
  address-family ipv4 unicast
    redistribute connected route-map RM-CONN-2-BGP
  exit-address-family
!
router bgp 65001
  bgp router-id 10.0.0.1
  bgp graceful-restart
  maximum-paths 4
  !
  address-family ipv4 unicast
    redistribute connected route-map RM-CONN-2-BGP
  exit-address-family
!

!
! Route-maps
route-map RM-CONN-2-BGP permit 10
  match interface Loopback0
!

!
! Management
line console
line vty
!
