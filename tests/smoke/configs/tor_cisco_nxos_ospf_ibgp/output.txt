! dc1-tor-01 Configuration

hostname dc1-tor-01
!
feature ospf
feature bgp
feature interface-vlan
feature lacp
feature vpc
feature lldp
!

! VLANs
vlan 100
  name Customer-A
!
vlan 200
  name Customer-A
!
!
!
! Interfaces
interface Ethernet1
  no shutdown
  ip address 10.1.0.1/31

  ip ospf area 0.0.0.0
!
interface Ethernet2
  no shutdown
  ip address 10.1.0.3/31

  ip ospf area 0.0.0.0
!
interface Ethernet10
  no shutdown

!
interface Loopback0
  no shutdown
  ip address 10.0.0.1/32

!
!
! OSPF Configuration

router ospf 1
  router-id 10.0.0.1
  auto-cost reference-bandwidth 100000
  area None authentication message-digest
  passive-interface default
!

!
! BGP Configuration

router bgp 65000
  router-id 10.0.0.1
  graceful-restart
  !
  template peer-policy EVPN-PEERS-POLICY
    send-community
    send-community extended
  !
  template peer-session EVPN-PEERS-SESSION
    update-source loopback0
  !
  address-family ipv4 unicast
    redistribute direct route-map RM-CONN-2-BGP
    maximum-paths 8
  !
  address-family l2vpn evpn
  !
  neighbor 10.0.0.100
    remote-as 65000
    description spine-01
    inherit peer-session EVPN-PEERS-SESSION
    address-family l2vpn evpn
      inherit peer-policy EVPN-PEERS-POLICY
    !
  neighbor 10.0.0.101
    remote-as 65000
    description spine-02
    inherit peer-session EVPN-PEERS-SESSION
    address-family l2vpn evpn
      inherit peer-policy EVPN-PEERS-POLICY
    !
!

!
!
! VXLAN Configuration (netlab-style multivendor)

feature nv overlay
feature vn-segment-vlan-based
feature nve

  member vni 10100
  no shutdown

vlan 100

route-map RM-CONN-2-BGP permit 10
