import * as aws from "@pulumi/aws";
import * as pulumi from "@pulumi/pulumi";

const aws_eu_central_1 = new aws.Provider("eu-central-1", { region: "eu-central-1" });

const prod_vpc = new aws.ec2.Vpc("prod-vpc", {
    cidrBlock: "10.0.0.0/16",
    enableDnsSupport: true,
    enableDnsHostnames: true,
    tags: { Name: "prod-vpc", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const subnet_prod_subnet_1a = new aws.ec2.Subnet("prod-subnet-1a", {
    vpcId: prod_vpc.id,
    cidrBlock: "10.0.1.0/24",
    availabilityZone: "eu-central-1a",
    mapPublicIpOnLaunch: false,
    tags: { Name: "prod-subnet-1a", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const sg_prod_sg = new aws.ec2.SecurityGroup("prod-sg", {
    vpcId: prod_vpc.id,
    tags: { Name: "prod-sg", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const instance_prod_instance_1 = new aws.ec2.Instance("prod-instance-1", {
    ami: "ami-12345678",
    instanceType: "t3.micro",
    subnetId: subnet_prod_subnet_1a.id,
    vpcSecurityGroupIds: [sg_prod_sg.id],
    tags: { Name: "prod-instance-1", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const igw_prod_igw = new aws.ec2.InternetGateway("prod-igw", {
    vpcId: prod_vpc.id,
    tags: { Name: "prod-igw", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const eip_prod_nat = new aws.ec2.Eip("prod-nat-eip", {
    domain: "vpc",
    tags: { Name: "prod-nat-eip", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const nat_prod_nat = new aws.ec2.NatGateway("prod-nat", {
    allocationId: eip_prod_nat.id,
    subnetId: subnet_prod_subnet_1a.id,
    tags: { Name: "prod-nat", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const rt_prod_rt = new aws.ec2.RouteTable("prod-rt", {
    vpcId: prod_vpc.id,
    tags: { Name: "prod-rt", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });

const route_route_igw = new aws.ec2.Route("route-igw", {
    routeTableId: rt_prod_rt.id,
    destinationCidrBlock: "0.0.0.0/0",
    gatewayId: igw_prod_igw.id,
}}, { provider: aws_eu_central_1 });

const rta_prod_subnet_1a_prod_rt = new aws.ec2.RouteTableAssociation("prod-subnet-1a-prod-rt-assoc", {
    subnetId: subnet_prod_subnet_1a.id,
    routeTableId: rt_prod_rt.id,
}}, { provider: aws_eu_central_1 });

const nacl_prod_nacl = new aws.ec2.NetworkAcl("prod-nacl", {
    vpcId: prod_vpc.id,
    subnetIds: [subnet_prod_subnet_1a.id],
    tags: { Name: "prod-nacl", ManagedBy: "Infrahub" },
}}, { provider: aws_eu_central_1 });
