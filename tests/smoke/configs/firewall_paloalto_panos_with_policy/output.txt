# --- Interfaces & Zone Assignments ---
set network interface ethernet ethernet1/1 layer3 units ethernet1/1.10 tag 10
set network interface ethernet ethernet1/1 layer3 units ethernet1/1.10 ipv4 addr 10.0.1.1/30
set network interface ethernet ethernet1/1.10 comment "Internal LAN link"
set zone internal network layer3 ethernet1/1.10
set network interface ethernet ethernet1/1 layer3 units ethernet1/1.20 tag 20
set network interface ethernet ethernet1/1 layer3 units ethernet1/1.20 ipv4 addr 10.0.2.1/30
set network interface ethernet ethernet1/1.20 comment "DMZ link"
set zone dmz network layer3 ethernet1/1.20

set network interface ethernet ethernet1/1 layer3 units ethernet1/1.30 ipv4 addr 10.0.3.1/30
set network interface ethernet ethernet1/1.30 comment "External link"
set zone external network layer3 ethernet1/1.30

#
# --- Address Objects (one per zone member CIDR) ---
set address addr-10-0-1-0-24 ip-netmask 10.0.1.0/24
set address addr-10-0-1-0-24 description "Zone: internal"
set address addr-10-0-10-0-24 ip-netmask 10.0.10.0/24
set address addr-10-0-10-0-24 description "Zone: internal"
set address addr-10-0-2-0-24 ip-netmask 10.0.2.0/24
set address addr-10-0-2-0-24 description "Zone: dmz"
#
# --- Address Groups (one per zone) ---
set address-group ag-internal static [ addr-10-0-1-0-24 addr-10-0-10-0-24 ]
set address-group ag-dmz static [ addr-10-0-2-0-24 ]
set address-group ag-external static [ any ]
#
#
# --- Static Routes (return path to leaf per zone) ---
set network virtual-router default routing-table ip static-route VRF-DMZ-1 destination 10.0.2.0/24
set network virtual-router default routing-table ip static-route VRF-DMZ-1 nexthop ip-address 10.0.2.2
set network virtual-router default routing-table ip static-route VRF-DMZ-1 interface ethernet1/1.20
set network virtual-router default routing-table ip static-route VRF-INTERNAL-2 destination 10.0.1.0/24
set network virtual-router default routing-table ip static-route VRF-INTERNAL-2 nexthop ip-address 10.0.1.2
set network virtual-router default routing-table ip static-route VRF-INTERNAL-2 interface ethernet1/1.10
set network virtual-router default routing-table ip static-route VRF-INTERNAL-3 destination 10.0.10.0/24
set network virtual-router default routing-table ip static-route VRF-INTERNAL-3 nexthop ip-address 10.0.1.2
set network virtual-router default routing-table ip static-route VRF-INTERNAL-3 interface ethernet1/1.10
